DataURL(base64)変換ツール

コードをクリップボードにコピーしました。

<script>
//DataURLへの変換コード
$("#file-input").on("change", e => { //ファイルアップロード時のイベント設定
    let file = e.target.files[0]; //ファイルの内容を取得
    if (!file) { //適応できない場合、プログラムを停止
        return;
    }

    let reader = new FileReader();  //FileReader APIの利用
    reader.onload = function(e){  //APIを発火する前に発火時の詳細をここで指定
        let base64 = e.target.result; //読み込み時の結果(result)を取得
        $("p.result").text(base64); //ブラウザに表示
        navigator.clipboard.writeText(base64); //クリップボードにコピー
    }
    reader.readAsDataURL(file);//実際にDataURLとしてファイルを読み込みを発火
})
</script>
