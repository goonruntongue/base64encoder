DataURL(base64)

<script>
//DataURL
$("#file-input").on("change", e => { //
    let file = e.target.files[0]; //
    if (!file) { //
        return;
    }

    let reader = new FileReader();  //FileReader API
    reader.onload = function(e){  //API
        let base64 = e.target.result; //(result)
        $("p.result").text(base64); //
        navigator.clipboard.writeText(base64); //
    }
    reader.readAsDataURL(file);//DataURL
})
</script>
